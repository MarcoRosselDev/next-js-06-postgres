# terminal comands

```
pwd = "/url/example"  

ls = lista de carpetas y archivos  

cd /url = cambia de directorio  

more <file> = visualiza el contenido de el archivo  

clear = limpia la terminal  

ls -l = lista en formato de lista   

mkdir nameFloder = crear una carpeta  

echo something = print something  

touch newFile = crea un archivo newFile.algo  

ls --all = muesta los archivos ++ocultos  

ls -a = muestra los archivos ++ocultos  

cp <file> <directorio> = copiar un file en un direcorio

rm <filename> = eliminar un archivo  

mv <filename> <new_filename> = renombrar un archivo  

mv <file> <destination> = mover un archivo a otro directorio  

find = lista el directorio en una estructura de arbol  

find -name index.html = buscar por todo el diretorio un archivo con ese nombre   

rmdir <nameFolder> = eliminar un directorio "carpeta"  

rm -r <name> = elimina cualquier cosa, incluida carpetas con su contenido, esto por que si la carpeta a eliminar continene algun archivo no ejecutara la eliminacion.


echo  I made this boilerplate >> README.md = imporimir el texto dentro de un archivo x
```
